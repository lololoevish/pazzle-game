```python
def setup_math_game(self):
    """Математическая мини-игра"""
    import random
    self.math_a = random.randint(1, 20)
    self.math_b = random.randint(1, 20)
    self.math_answer = self.math_a + self.math_b
    self.math_input = ""
    
def update_math_game(self):
    """Обновление математической игры"""
    if self.game_timer > 0:
        self.game_timer -= 1
    else:
        self.result = 'lose'
        self.game_complete = True
        
def handle_math_input(self, event):
    """Обработка ввода для математики"""
    if event.type == pygame.KEYDOWN:
        if event.key == pygame.K_RETURN:
            try:
                if int(self.math_input) == self.math_answer:
                    self.result = 'win'
                else:
                    self.result = 'lose'
                self.game_complete = True
            except:
                pass
        elif event.key == pygame.K_BACKSPACE:
            self.math_input = self.math_input[:-1]
        elif event.unicode.isdigit():
            self.math_input += event.unicode
```

Затем в методе `__init__()` добавить:

### 14. Добавить систему достижений

**Файл:** `src/utils/achievements.py` (создать новый)

```python
class AchievementSystem:
    def __init__(self):
        self.achievements = {
            'first_level': {'name': 'Первый шаг', 'unlocked': False},
            'all_levels': {'name': 'Мастер головоломок', 'unlocked': False},
            'rich': {'name': 'Богач', 'unlocked': False},  # 1000 золота
            'collector': {'name': 'Коллекционер', 'unlocked': False}  # 10 предметов
        }
        
    def check_achievement(self, achievement_id, condition):
        """Проверить и разблокировать достижение"""
        if achievement_id in self.achievements and condition:
            if not self.achievements[achievement_id]['unlocked']:
                self.achievements[achievement_id]['unlocked'] = True
                return True  # Достижение только что разблокировано
        return False
        
    def get_unlocked_count(self):
        """Получить количество разблокированных достижений"""
        return sum(1 for a in self.achievements.values() if a['unlocked'])
```

```python
import pygame
import random
import math

class VictoryParticles:
    def __init__(self, x, y):
        self.particles = []
        for _ in range(50):
            angle = random.uniform(0, 2 * math.pi)
            speed = random.uniform(2, 8)
            self.particles.append({
                'x': x,
                'y': y,
                'vx': math.cos(angle) * speed,
                'vy': math.sin(angle) * speed,
                'life': random.randint(30, 60),
                'color': random.choice([
                    (255, 215, 0),
                    (255, 255, 0),
                    (255, 165, 0)
                ]),
                'size': random.randint(3, 8)
            })
            
    def update(self):
        """Обновить частицы"""
        for particle in self.particles[:]:
            particle['x'] += particle['vx']
            particle['y'] += particle['vy']
            particle['vy'] += 0.3  # Гравитация
            particle['life'] -= 1
            if particle['life'] <= 0:
                self.particles.remove(particle)
                
    def draw(self, screen):
        """Отрисовать частицы"""
        for particle in self.particles:
            alpha = int(255 * (particle['life'] / 60))
            surf = pygame.Surface((particle['size'] * 2, particle['size'] * 2), 
                                 pygame.SRCALPHA)
            color = (*particle['color'], alpha)
            pygame.draw.circle(surf, color, 
                             (particle['size'], particle['size']), 
                             particle['size'])
            screen.blit(surf, (int(particle['x']), int(particle['y'])))
            
    def is_finished(self):
        """Проверить, закончились ли частицы"""
        return len(self.particles) == 0
```

### 16. Добавить эффект тряски экрана

**Файл:** `src/utils/visual_effects.py`

```python
import random

class ScreenShake:
    def __init__(self, intensity=10, duration=20):
        self.intensity = intensity
        self.duration = duration
        self.timer = 0
        
    def start(self):
        """Запустить тряску"""
        self.timer = self.duration
        
    def update(self):
        """Обновить тряску"""
        if self.timer > 0:
            self.timer -= 1
            
    def get_offset(self):
        """Получить смещение для камеры"""
        if self.timer > 0:
            shake_x = random.randint(-self.intensity, self.intensity)
            shake_y = random.randint(-self.intensity, self.intensity)
            return shake_x, shake_y
        return 0, 0
        
    def is_shaking(self):
        """Проверить, идёт ли тряска"""
        return self.timer > 0
```
